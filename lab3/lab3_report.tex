\subsection{中序表达式到后序表达式的实验结果}

\begin{enumerate}
	\item 
Infix:( A + B ) * C

Postfix:A B + C *

\item
Infix:A + ( B - C )

Postfix:A B C - +

\item
Infix:A * ( B + C ) / D

Postfix:A B C + * D /

\item
Infix:( A + B ) * ( C - D )

Postfix:A B + C D - *

\item 
Infix: A + B * C - D / E

Postfix:A B C * + D E / -

\item
Infix:( A * B ) + ( C / D ) - E

Postfix:A B * C D / + E -

\item
Infix:( A + B ) / ( C + D ) * E

Postfix:A B + C D + / E *

\item
Infix: A * ( B + C ) - ( D * E )

Postfix:A B C + * D E * -

\item
Infix:( A + B ) * ( C - D ) / ( E + F )

Postfix:A B + C D - * E F + /

\item
Infix: A * ( B + ( C * ( D - ( E / ( F + ( G * H ) ) ) ) ) )  / I

Postfix:A B C D E F G H * + / - * + * I /

\end{enumerate}
